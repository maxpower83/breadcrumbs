       // Procedural room generators
       const ROOM_PREFIXES = ['Biblioteca', 'Cocina', 'Salón', 'Invernadero', 'Estudio', 'Comedor', 'Dormitorio', 'Sótano', 'Ático', 'Vestíbulo', 'Baño', 'Oficina', 'Terraza', 'Despensa', 'Garaje', 'Galería', 'Pasillo', 'Búnker', 'Cripta', 'Mirador', 'Solárium', 'Gimnasio', 'Teatro', 'Laboratorio'];
       const ROOM_ADJECTIVES = ['Imperial', 'Sombrío', 'Antiguo', 'Moderno', 'Secreto', 'Occidental', 'Oriental', 'Abandonado', 'Principal', 'Siniestro', 'Acogedor', 'Polvoriento', 'Lujoso', 'Gótico', 'Olvidado', 'Prohibido'];

       function generateProceduralRooms(size) {
           const count = size * size;
           let rooms = [];
           let unusedPrefixes = shuffle([...ROOM_PREFIXES]);
           let unusedAdjectives = shuffle([...ROOM_ADJECTIVES]);

           for (let i = 0; i < count; i++) {
               if (unusedPrefixes.length === 0) unusedPrefixes = shuffle([...ROOM_PREFIXES]);
               if (unusedAdjectives.length === 0) unusedAdjectives = shuffle([...ROOM_ADJECTIVES]);

               const prefix = unusedPrefixes.pop();
               const adj = unusedAdjectives.pop();
               
               rooms.push(`${prefix} ${adj}`);
           }
           return rooms;
       }

       function shuffle(arr) {
           const newArr = [...arr];
           for (let i = newArr.length - 1; i > 0; i--) {
               const j = Math.floor(Math.random() * (i + 1));
               [newArr[i], newArr[j]] = [newArr[j], newArr[i]];
           }
           return newArr;
       }

       function generateSolution() {
           const N = gameState.size;
           let suspectsGrid = Array(N * N).fill(null);
           let weaponsGrid = Array(N * N).fill(null);

           gameState.suspects = ALL_SUSPECTS.slice(0, N);
           gameState.weapons = ALL_WEAPONS.slice(0, N);
           gameState.rooms = generateProceduralRooms(N);

           let sCols = [];
           for (let i = 0; i < N; i++) sCols.push(i);
           sCols = shuffle(sCols);

           for (let r = 0; r < N; r++) {
               const sId = gameState.suspects[r].id;
               const col = sCols[r];
               suspectsGrid[r * N + col] = sId;
           }

           let wCols;
           let attempts = 0;
           while (attempts < 3000) {
               wCols = [];
               for (let i = 0; i < N; i++) wCols.push(i);
               wCols = shuffle(wCols);

               let intersections = 0;
               for (let r = 0; r < N; r++) {
                   if (sCols[r] === wCols[r]) {
                       intersections++;
                   }
               }
               if (intersections === 1) {
                   break;
               }
               attempts++;
           }

           const shuffledWeaponIds = shuffle(gameState.weapons.map(w => w.id));
           for (let r = 0; r < N; r++) {
               const wId = shuffledWeaponIds[r];
               const col = wCols[r];
               weaponsGrid[r * N + col] = wId;
           }

           gameState.solution.suspects = suspectsGrid;
           gameState.solution.weapons = weaponsGrid;

           for (let i = 0; i < N * N; i++) {
               if (suspectsGrid[i] && weaponsGrid[i]) {
                   gameState.crimeSceneIndex = i;
                   gameState.murdererId = suspectsGrid[i];
                   gameState.murderWeaponId = weaponsGrid[i];
                   break;
               }
           }
       }

       function generateClues() {
           const N = gameState.size;
           const sGrid = gameState.solution.suspects;
           const wGrid = gameState.solution.weapons;
           const crimeIdx = gameState.crimeSceneIndex;

           let cluesPool = [];

           const getPos = (index) => ({
               r: Math.floor(index / N) + 1,
               c: (index % N) + 1,
               room: gameState.rooms[index]
           });

           let suspectLocations = {};
           let weaponLocations = {};
           for(let i = 0; i < N * N; i++) {
               if(sGrid[i]) suspectLocations[sGrid[i]] = i;
               if(wGrid[i]) weaponLocations[wGrid[i]] = i;
           }

           gameState.suspects.forEach(s => {
               const idx = suspectLocations[s.id];
               if (idx !== crimeIdx) {
                   const pos = getPos(idx);
                   cluesPool.push({
                       text: `Una huella digital asocia a <strong>${s.name}</strong> directamente con la habitación: <strong>${pos.room}</strong>.`,
                       category: 'direct'
                   });
               }
           });

           gameState.weapons.forEach(w => {
               const idx = weaponLocations[w.id];
               if (idx !== crimeIdx) {
                   const pos = getPos(idx);
                   cluesPool.push({
                       text: `Se ha hallado una funda vacía para el arma <strong>${w.name}</strong> en la habitación: <strong>${pos.room}</strong>.`,
                       category: 'direct'
                   });
               }
           });

           gameState.suspects.forEach(s => {
               gameState.weapons.forEach(w => {
                   const sIdx = suspectLocations[s.id];
                   const wIdx = weaponLocations[w.id];
                   
                   if (sIdx === crimeIdx && wIdx === crimeIdx) return;

                   const sPos = getPos(sIdx);
                   const wPos = getPos(wIdx);

                   if (sPos.r === wPos.r && sIdx !== wIdx) {
                       cluesPool.push({
                           text: `Un plano estructural revela que la habitación de <strong>${s.name}</strong> comparte la misma <strong>Fila</strong> que la habitación con la <strong>${w.name}</strong>.`,
                           category: 'alignment'
                       });
                   }
                   if (sPos.c === wPos.c && sIdx !== wIdx) {
                       cluesPool.push({
                           text: `Un informe del servicio doméstico alinea la columna de <strong>${s.name}</strong> con la columna donde se encuentra la <strong>${w.name}</strong>.`,
                           category: 'alignment'
                       });
                   }
               });
           });

           gameState.suspects.forEach(s => {
               gameState.weapons.forEach(w => {
                   const sIdx = suspectLocations[s.id];
                   const wIdx = weaponLocations[w.id];
                   if (sIdx === crimeIdx && wIdx === crimeIdx) return;

                   const sPos = getPos(sIdx);
                   const wPos = getPos(wIdx);

                   if (sPos.r < wPos.r) {
                       cluesPool.push({
                           text: `La habitación de <strong>${s.name}</strong> se sitúa en una fila <strong>superior</strong> (norte) respecto a la <strong>${w.name}</strong>.`,
                           category: 'relative'
                       });
                   } else if (sPos.r > wPos.r) {
                       cluesPool.push({
                           text: `La habitación de <strong>${s.name}</strong> se sitúa en una fila <strong>inferior</strong> (sur) respecto a la <strong>${w.name}</strong>.`,
                           category: 'relative'
                       });
                   }

                   if (sPos.c < wPos.c) {
                       cluesPool.push({
                           text: `La columna de <strong>${s.name}</strong> está a la <strong>izquierda</strong> (oeste) de la columna donde reposa la <strong>${w.name}</strong>.`,
                           category: 'relative'
                       });
                   } else if (sPos.c > wPos.c) {
                       cluesPool.push({
                           text: `La columna de <strong>${s.name}</strong> está a la <strong>derecha</strong> (este) de la columna donde reposa la <strong>${w.name}</strong>.`,
                           category: 'relative'
                       });
                   }
               });
           });

           let emptyCount = 0;
           const maxEmptyClues = N === 4 ? 4 : N === 6 ? 6 : 8;
           for (let i = 0; i < N * N; i++) {
               if (!sGrid[i] && !wGrid[i]) {
                   cluesPool.push({
                       text: `El oficial de guardia examinó la habitación <strong>${gameState.rooms[i]}</strong> y está totalmente desocupada.`,
                       category: 'negative'
                   });
                   emptyCount++;
                   if (emptyCount >= maxEmptyClues) break;
               }
           }

           let finalClues = [];
           let directClues = cluesPool.filter(c => c.category === 'direct');
           let relationalClues = cluesPool.filter(c => c.category !== 'direct');

           shuffle(directClues);
           shuffle(relationalClues);

           let limitDirect = 2;
           let limitRelational = 6;

           if (N === 4) {
               limitDirect = 2;
               limitRelational = 5;
           } else if (N === 6) {
               limitDirect = 3;
               limitRelational = 8;
           } else {
               limitDirect = 4;
               limitRelational = 12;
           }

           finalClues.push(...directClues.slice(0, limitDirect));
           finalClues.push(...relationalClues.slice(0, limitRelational));

           gameState.clues = shuffle(finalClues);
       }

       // --- VERIFY ACCUSATION ---

       function verifyAccusation() {
           if (gameState.gameOver) return;

           const accusedSuspect = document.getElementById('accuse-suspect').value;
           const accusedWeapon = document.getElementById('accuse-weapon').value;
           const accusedRoomIndex = document.getElementById('accuse-room').value;

           if (!accusedSuspect || !accusedWeapon || accusedRoomIndex === "") {
               playSound('fail');
               showVerdictModal(
                   'fail',
                   'Expediente Incompleto',
                   'Debes señalar un sospechoso, un arma homicida y la habitación del crimen antes de presentar los cargos ante el Juez.',
                   false
               );
               return;
           }

           const isCorrect = (accusedSuspect === gameState.murdererId) &&
                             (accusedWeapon === gameState.murderWeaponId) &&
                             (parseInt(accusedRoomIndex) === gameState.crimeSceneIndex);

           if (isCorrect) {
               gameState.gameOver = true;
               const murderer = gameState.suspects.find(s => s.id === gameState.murdererId);
               const weapon = gameState.weapons.find(w => w.id === gameState.murderWeaponId);
               const roomName = gameState.rooms[gameState.crimeSceneIndex];

               playSound('success');
               showVerdictModal(
                   'success',
                   '¡Caso Resuelto con Éxito!',
                   `Has deducido con precisión quirúrgica el sumario judicial.`,
                   true,
                   `Tras recabar pruebas lógicas incontestables, se condena formalmente a <strong>${murderer.name} (${murderer.icon})</strong> por el cargo de homicidio premeditado en la habitación <strong>${roomName}</strong> empleando como arma mortífera la <strong>${weapon.name} (${weapon.icon})</strong>.`
               );
               renderBoard();
           } else {
               gameState.attemptsRemaining--;
               renderAttempts();

               if (gameState.attemptsRemaining <= 0) {
                   gameState.gameOver = true;
                   playSound('fail');
                   
                   const actualMurderer = gameState.suspects.find(s => s.id === gameState.murdererId);
                   const actualWeapon = gameState.weapons.find(w => w.id === gameState.murderWeaponId);
                   const actualRoom = gameState.rooms[gameState.crimeSceneIndex];

                   showVerdictModal(
                       'escape',
                       '¡El Asesino ha Escapado!',
                       `Demasiados errores de investigación. El asesino se sintió acorralado por tus acusaciones fallidas y huyó de la mansión.`,
                       true,
                       `El expediente queda archivado sin resolver. Las pruebas forenses posteriores confirman que el autor real era <strong>${actualMurderer.name} (${actualMurderer.icon})</strong> en la habitación <strong>${actualRoom}</strong> con la <strong>${actualWeapon.name} (${actualWeapon.icon})</strong>.`
                   );
                   renderBoard();
               } else {
                   playSound('fail');
                   showVerdictModal(
                       'fail',
                       'Acusación Errónea',
                       `El sospechoso tiene coartada o el arma no coincide con la autopsia. Tus movimientos erráticos alertan al asesino. ¡Te quedan ${gameState.attemptsRemaining} intentos!`,
                       false
                   );
               }
           }
       }

       }

       // --- GAME CONTROLLER ---

       function initNewGame() {
           const size = parseInt(document.getElementById('grid-size').value);
           gameState.size = size;
           gameState.gameOver = false;

           // Scaled attempts based on board size
           if (size === 4) {
               gameState.maxAttempts = 2;
           } else if (size === 6) {
               gameState.maxAttempts = 3;
           } else {
               gameState.maxAttempts = 4;
           }
           gameState.attemptsRemaining = gameState.maxAttempts;

           document.getElementById('mansion-size-text').textContent = size * size;
           document.getElementById('elements-count-text').textContent = size;
           document.getElementById('weapons-count-text').textContent = size;

           gameState.player.suspects = Array(size * size).fill(null);
           gameState.player.weapons = Array(size * size).fill(null);
           gameState.player.notes = Array(size * size).fill(null).map(() => ({
               suspects: {},
               weapons: {}
           }));

           generateSolution();
           generateClues();

           renderBoard();
           renderCluesList();
           renderAttempts();
           populateAccusationDropdowns();
       }

               playSound('click');
               initNewGame();
           });

           document.getElementById('btn-clear-board').addEventListener('click', () => {
               if (gameState.gameOver) return;
               playSound('click');
               for(let i = 0; i < gameState.size * gameState.size; i++) clearCell(i);
               renderBoard();
           });

           document.getElementById('btn-submit-accusation').addEventListener('click', () => {
               playSound('click');
               verifyAccusation();
           });

           document.getElementById('grid-size').addEventListener('change', () => {
               playSound('click');
               initNewGame();
           });

           document.getElementById('tab-assign').addEventListener('click', () => {
               playSound('click');
               switchTab('assign');
           });
           document.getElementById('tab-notes').addEventListener('click', () => {
               playSound('click');
               switchTab('notes');
           });

           document.getElementById('modal-close').addEventListener('click', () => {
               playSound('click');
               closeEditorModal();
               renderBoard();
           });
           document.getElementById('modal-save').addEventListener('click', () => {
               playSound('click');
               closeEditorModal();
               renderBoard();
           });
           document.getElementById('modal-clear-cell').addEventListener('click', () => {

               closeVerdictModal();

               playSound('click');
               closeVerdictModal();
               initNewGame();
           });
